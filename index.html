// استخدم رابط Railway بدلاً من Supabase مباشرة const API_BASE_URL = 'https://real-team.up.railway.app'; // جلب الأرباح async function loadEarnings() { try { const response = await fetch(`${API_BASE_URL}/api/earnings`); const data = await response.json(); if (data.success) { displayEarnings(data.data); updateStats(data.data); } } catch (error) { console.error('Error:', error); } } // إضافة ربح جديد async function addEarning() { const platform = document.getElementById('platform').value; const amount = parseFloat(document.getElementById('amount').value); const source = document.getElementById('source').value; const status = document.getElementById('statusSelect').value; try { const response = await fetch(`${API_BASE_URL}/api/earnings`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ platform, amount, source, status }) }); const result = await response.json(); if (result.success) { alert(`✅ تم إضافة ${amount} درهم`); loadEarnings(); } } catch (error) { alert('❌ خطأ: ' + error.message); } }
